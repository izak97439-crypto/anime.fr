        function createCard(movie, options = {}) {
            const card = document.createElement('div');
            card.className = 'card';
            card.setAttribute('data-movie-id', movie.id);
            card.addEventListener('mouseenter', () => { if (movie.tmdbId) prefetchVideo(movie.tmdbId, 1); });
            card.onclick = () => openModal(movie.id);
            const img = document.createElement('img');
            img.className = 'card-poster';
            img.src = (movie.posterSeed && movie.posterSeed.startsWith('http')) ? movie.posterSeed : `https://picsum.photos/seed/${movie.posterSeed}/440/660`;
            img.alt = movie.title;
            img.loading = 'lazy';
            card.appendChild(img);
            if (options.showProgress && typeof options.progress === 'number') {
                const prog = document.createElement('div'); prog.className = 'card-progress';
                const fill = document.createElement('div'); fill.className = 'card-progress-fill';
                fill.style.width = `${options.progress}%`; prog.appendChild(fill); card.appendChild(prog);
            }
            if (options.showTop10 && movie.isTop10 && movie.top10Rank) {
                const badge = document.createElement('span'); badge.className = 'card-top10';
                badge.textContent = movie.top10Rank; card.appendChild(badge);
            }
            const info = document.createElement('div'); info.className = 'card-info';
            info.innerHTML = `<div class="card-info-title">${movie.title}</div><div class="card-info-meta"><span class="match-badge">${movie.match}%</span><span>${movie.year}</span><span>${movie.rating}</span></div>`;
            card.appendChild(info);
            return card;
        }

        function createRow(category) {
            const section = document.createElement('section');
            section.className = 'content-section';
            section.setAttribute('data-category-id', category.id);
            if (category.id !== 'continue') {
                const header = document.createElement('div');
                header.className = 'section-header';
                header.innerHTML = `<h2 class="section-title">${category.title}</h2><a href="javascript:void(0);" class="section-explore">Explorer tout ›</a>`;
                section.appendChild(header);
            }
            const wrapper = document.createElement('div'); wrapper.className = 'row-wrapper';
            const row = document.createElement('div'); row.className = 'row'; row.setAttribute('data-row-id', category.id);
            if (category.items) {
                category.items.forEach(item => { const movie = getMovieById(item.movieId); if (movie) row.appendChild(createCard(movie, { showProgress: true, progress: item.progress })); });
            } else if (category.movieIds) {
                category.movieIds.forEach(movieId => { const movie = getMovieById(movieId); if (movie) row.appendChild(createCard(movie, { showTop10: category.showTop10 || false })); });
            }
            const btnLeft = document.createElement('button'); btnLeft.className = 'scroll-btn left hidden'; btnLeft.innerHTML = '‹';
            btnLeft.onclick = () => scrollRow(row, -500);
            const btnRight = document.createElement('button'); btnRight.className = 'scroll-btn right'; btnRight.innerHTML = '›';
            btnRight.onclick = () => scrollRow(row, 500);
            wrapper.appendChild(row); wrapper.appendChild(btnLeft); wrapper.appendChild(btnRight); section.appendChild(wrapper);
            updateScrollButtons(row, btnLeft, btnRight);
            row.addEventListener('scroll', () => updateScrollButtons(row, btnLeft, btnRight));
            window.addEventListener('resize', () => updateScrollButtons(row, btnLeft, btnRight));
            return section;
        }
        
        function scrollRow(row, amount) { row.scrollBy({ left: amount, behavior: 'smooth' }); }
        
        function updateScrollButtons(row, btnLeft, btnRight) {
            const isAtStart = row.scrollLeft <= 5;
            const isAtEnd = row.scrollLeft + row.clientWidth >= row.scrollWidth - 5;
            btnLeft.classList.toggle('hidden', isAtStart);
            btnRight.classList.toggle('hidden', isAtEnd);
        }

                li.textContent = `${genre.icon} ${genre.name}`;
                li.addEventListener('click', (e) => {

                    searchInput.value = genre.name;
                    performSearch(genre.name);
                });
                dropdownList.appendChild(li);
            });
            const separator = document.createElement('li');
            separator.style.borderTop = '1px solid #333'; separator.style.margin = '5px 0'; separator.style.padding = '0'; separator.style.cursor = 'default';

                li.addEventListener('click', (e) => { e.stopPropagation(); openModal(movie.id); });
                dropdownList.appendChild(li);
            });
        }

        function performSearch(query) {
            const lowerQuery = query.toLowerCase();
            const filtered = movies.filter(m => m.title.toLowerCase().includes(lowerQuery) || m.cast.toLowerCase().includes(lowerQuery) || m.genre.toLowerCase().includes(lowerQuery));
            searchRow.innerHTML = '';
            searchTitle.textContent = filtered.length === 0 ? 'Aucun résultat' : `Résultats de recherche (${filtered.length})`;
            filtered.forEach(m => searchRow.appendChild(createCard(m, { showTop10: false })));
            contentRowsContainer.style.display = 'none';
            searchResults.classList.add('active');
        }

        function getIframeUrl(serverIndex, tmdbId, episode, season) {
            season = season || currentSeason || 1;
            const slug = animeSamaSlugs[tmdbId];
            switch(serverIndex) {
                case 0:
                    // Anime-Sama : meilleure source VF francophone
                    if (slug) return `https://anime-sama.fr/catalogue/${slug}/saison${season}/vf/episode${episode}/`;
                    return `https://anime-sama.fr/`;
                case 1:
                    // Neko-Sama : alternative VF
                    if (slug) return `https://www.neko-sama.fr/anime/info/${slug}_VF`;
                    return `https://www.neko-sama.fr/`;
                case 2:
                    // Franime / vidsrc fallback
                    return `https://vidsrc.me/embed/tv?tmdb=${tmdbId}&season=${season}&episode=${episode}&autoplay=1`;
                default:
                    if (slug) return `https://anime-sama.fr/catalogue/${slug}/saison${season}/vf/episode${episode}/`;
                    return `https://vidsrc.me/embed/tv?tmdb=${tmdbId}&season=${season}&episode=${episode}&autoplay=1`;
            }
        }

        function prefetchVideo(tmdbId, episode) {
            if (prefetchFrame) { prefetchFrame.remove(); prefetchFrame = null; }
            prefetchFrame = document.createElement('iframe');
            prefetchFrame.style.cssText = 'position:absolute;width:1px;height:1px;opacity:0;pointer-events:none;border:none;';

            prefetchFrame.src = pfSlug
                ? `https://anime-sama.fr/catalogue/${pfSlug}/saison1/vf/episode${episode}/`
                : `https://vidsrc.me/embed/tv?tmdb=${tmdbId}&season=1&episode=${episode}&autoplay=1`;
            document.body.appendChild(prefetchFrame);
        }

        function playVideo(url) {
            const videoLoading = document.getElementById('videoLoading');
            const fsEl = document.getElementById('modalMediaContainer');

            // Si l'iframe prefetch correspond à l'URL demandée, la réutiliser
            modalPoster.style.display = 'none';
            modalVideoPlayer.style.display = 'block';

            // Swap immédiat : on affiche l'iframe sans attendre
            if (modalVideoPlayer.src !== url) {
                videoLoading.classList.add('active');
                modalVideoPlayer.src = url;
                // Masquer le spinner rapidement dès que l'iframe répond
                modalVideoPlayer.onload = () => videoLoading.classList.remove('active');
                // Fallback : forcer masquage après 2s max
                setTimeout(() => videoLoading.classList.remove('active'), 2000);
            } else {
                videoLoading.classList.remove('active');
            }

            // Plein écran automatique
            if (fsEl.requestFullscreen) fsEl.requestFullscreen();
            else if (fsEl.webkitRequestFullscreen) fsEl.webkitRequestFullscreen();
            else if (fsEl.mozRequestFullScreen) fsEl.mozRequestFullScreen();

            if (window.innerWidth <= 900) fsEl.scrollIntoView({ behavior: 'smooth' });

            // Supprimer le frame de prefetch s'il existe
            if (prefetchFrame) { prefetchFrame.remove(); prefetchFrame = null; }
        }

        // Cache global Kitsu : title → { poster, cover, tiny }
        const kitsuCache = {};

        async function fetchKitsuData(title) {
            if (kitsuCache[title]) return kitsuCache[title];
            try {
                const response = await fetch(`https://kitsu.io/api/edge/anime?filter[text]=${encodeURIComponent(title)}&page[limit]=1`);
                const data = await response.json();
                if (data.data && data.data.length > 0) {
                    const attr = data.data[0].attributes;
                    const result = {
                        poster: (attr.posterImage && (attr.posterImage.medium || attr.posterImage.small)) || null,
                        cover:  (attr.coverImage  && (attr.coverImage.large  || attr.coverImage.original || attr.coverImage.small)) || null,
                        tiny:   (attr.posterImage && attr.posterImage.tiny) || null
                    };
                    kitsuCache[title] = result;
                    return result;
                }
            } catch (e) { console.warn(`Kitsu fetch error: ${title}`); }
            return null;
        }
